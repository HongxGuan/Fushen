int arrinput(int arr[]) {
	int len = 0;
	do {
		scanf("%d", arr + (len++));
	} while (getchar() != '\n');
	return len;
}

void arroutput(int arr[], int len) {
	int i;
	for (i = 0; i < len; i++) {
		printf("%d ", arr[i]);
	}
	putchar('\n');
	return;
}

double average(int arr[], int len) {
	double sum = 0.0;
	int i;
	for (i = 0; i < len; i++) {
		sum += arr[i];
	}
	double ave = sum / len;
	return ave;
}

int MIN(int arr[], int len) {
	int min = arr[0];
	int i;
	for (i = 1; i < len; i++) {
		if (arr[i] < min)
			min = arr[i];
	}
	return min;
}

int MAX(int arr[], int len) {
	int max = arr[0];
	int i;
	for (i = 1; i < len; i++) {
		if (arr[i] > max)
			max = arr[i];
	}
	return max;
}

void is_itemexist(int arr[], int item, int len) {
	int *index = (int *)malloc(len * sizeof(int));
	int i, j = 0;
	for (i = 0; i < len; i++) {
		if (arr[i] == item)
			index[j++] = i;
	}
	if (j == 0) {
		printf("Item %d NOT exist in this array!", item);
		putchar('\n');
	}
	else {
		printf("Item %d EXIST! The index is :", item);
		arroutput(index, j);
	}
	return;
}

void sort(int arr[], int len) {
	int i, j;
	int index = 0;
	int temp;
	for (i = 0; i < len; i++) {
		index = i;
		for (j = i; j < len; j++) {
			if (arr[j] < arr[index])
				index = j;
		}
		if (index != i) {
			temp = arr[index];
			arr[index] = arr[i];
			arr[i] = temp;
		}
	}
	return;
}

void arrcpy(int arr[], int arr_s[], int len) {
	int i;
	for (i = 0; i < len; i++) {
		arr[i] = arr_s[i];
	}
	return;
}

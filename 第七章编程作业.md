- ###### 1.6_20交换排序

  ```c
  #include<stdio.h>
  int main() {
  	int arr[10] = { 0 };
  	int n;
  	scanf("%d", &n);
  	getchar();
  	int i;
  	for (i = 0; i < n; i++) {
  		scanf("%d", &arr[i]);
  		getchar();
  	}
  	int times, j;
  	int min, flag, check;
  	for (times = 0; times < n - 1; times++) {
  		check = 0;
  		min = arr[times];
  		for (j = times + 1; j <= n - 1; j++) {
  			if (arr[j] < min) {
  				min = arr[j];
  				flag = j;
  				check = 1;
  			}
  		}
  		if (check) {
  			arr[flag] = arr[times];
  			arr[times] = min;
  		}
  	}
  	int k;
  	for (k = 0; k < n; k++) {
  		printf("%d ", arr[k]);
  	}
  	getchar();
  	return 0;
  }
  ```

- ###### 2.7-4-2 寻找矩阵中的马鞍点

  ```c
  #include<stdio.h>
  #define ROW 3
  #define COL 5
  int is_rowmin(int row, int col, int arr[ROW][COL]);
  int is_colmax(int row, int col, int arr[ROW][COL]);
  int main() {
  	int arr[3][5] = { { 6,3,5,3,4 },{ 4,3,4,3,4 },{ 5,2,4,2,5 } };
  	int i, j;
  	for (i = 0; i < ROW; i++) {
  		for (j = 0; j < COL; j++) {
  			if (is_rowmin(i, j, arr) && is_colmax(i, j, arr)) {
  				printf("(%3d)", arr[i][j]);
  			}
  			else {
  				printf("%5d", arr[i][j]);
  			}
  		}
  		putchar('\n');
  	}
  	getchar();
  	return 0;
  }
  
  int is_rowmin(int row, int col, int arr[ROW][COL]) {
  	int i;
  	int flag = 1;
  	for (i = 0; i < COL; i++) {
  		if (arr[row][i] < arr[row][col]) {
  			flag = 0;
  			break;
  		}
  	}
  	return flag;
  }
  int is_colmax(int row, int col, int arr[ROW][COL]) {
  	int i;
  	int flag = 1;
  	for (i = 0; i < ROW; i++) {
  		if (arr[i][col] > arr[row][col]) {
  			flag = 0;
  			break;
  		}
  	}
  	return flag;
  }
  ```

- ###### 3.7-4-5 数组元素循环移动

  ```c
  #include<stdio.h>
  
  void roll(int n, int arr[]);
  
  int main() {
  	int n, m, arr[20] = { 0 };
  	scanf("%d", &n);
  	getchar();
  	int i;
  	for (i = 1; i <= n; i++) {
  		scanf("%d", &arr[i]);
  		getchar();
  	}
  	scanf("%d", &m);
  	getchar();
  	int j;
  	for (j = 0; j < m; j++) {
  		roll(n, arr);
  	}
  	int k;
  	for (k = 1; k <= n; k++) {
  		printf("%d ", arr[k]);
  	}
  	getchar();
  	return 0;
  }
  
  void roll(int n, int arr[]) {
  	arr[0] = arr[n];
  	int i;
  	for (i = n; i >= 1; i--) {
  		arr[i] = arr[i - 1];
  	}
  }
  ```
